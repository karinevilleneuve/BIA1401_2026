bacteria = c(“Escherichia;coli”, “Enterobacter;cloacae”, “Staphylococcus;epidermidis”, “Citrobacter;youngae”,“Serratia;fonticola”, “Buttiauxella;agrestis”, “Bacillus;subtilis”, “Acinetobacter;calcoaceticus”) pattern_regex = paste(bacteria, collapse = “|”)

dna_sequences = Biostrings::readDNAStringSet(“silva_nr99_v138_2_BIA1401.fasta”) head(names(dna_sequences))

df = data.frame(name = names(dna_sequences),sequence = as.character(dna_sequences), Sequence_lenght = width(dna_sequences))

nos_sequences = df %>% mutate(match = str_match(name, pattern_regex)) %>% drop_na()

write.fasta(as.list(nos_sequences\(sequence), names = nos_sequences\)name, nbchar = 80, as.string = TRUE, file.out = “sequence_to_align.fasta”) ```
